(define-syntax trace
  (syntax-rules ()
    ((trace value)
     (begin
       (let ((val value))
         (write 'value)
         (display " => ")
         (write val)
         (newline)
         val)))))


(define (zip . xss)
  (if (or (null? xss)
          (null? (trace (car xss))))
      '()
      (cons (map car xss)
            (apply zip (map cdr(trace xss))))))


(define-syntax test
  (syntax-rules ()
    ((test expr result)
     (list (lambda () expr)
           'result 'expr))))

(define (signum x)
  (cond

    ((= x 0) 1)

(define (run-test xs)
  (write (caddr xs))
  (let ((eqv ((car xs))))
    (if (equal? (cadr xs) eqv)
        (begin  
          (display " ok")
          (newline)
          #t)
     
        (begin
          (display " FAIL")
          (newline)
          (display "Expected: ")
          (write (cadr xs))
          (newline)
          (display "Returned: ")
          (write eqv)
          (newline)
          #f
          ))))
    

(define (run-tests xss)
  (define (loop xs yn)
    (if (not (null? xs))
        (let ((rs (run-test (car xs))))
          (loop (cdr xs) (and yn rs)))
        yn))
  (let ((res (loop xss #t)))
    res))


(define (ref xss idx . new)

  (define (loop xs id)

             (if (null? new)
                 (car xs)
                 (append (reverse xs) new))
             (let ((r (loop (cdr xs) (- id 1))))
               (if (pair? r)
                   (append r (cons (car xs) '()))
                   (append (list r)))))))
        

  (cond
    ((string? xss) (and (or (null? new)
                            (char? (car new)))

                                   (if (pair? res)
                                       (car res)
                                       res)
                                   (list->string (reverse res)))))))
    ((vector? xss) (let ((res (loop (vector->list xss) idx)))
                     (and res
                          (if (null? new)
                              (if (pair? res)
                                  (car res)
                                  res)
                              (list->vector (reverse res))))))
    ((list? xss) (let ((res (loop xss idx)))
                   (and res
                        (if (null? new)
                            (if (pair? res)
                                (car res)
                                res)
                            (reverse res)))))))



(define (if->cond expr)
  (define (expand e)
    (if (and (list? e) (eq? (car e) 'if))
        (let ((test (cadr e))
              (then (caddr e))
              (else-part (if (>= (length e) 4) (cadddr e) #f)))
          (if (and (list? else-part) (eq? (car else-part) 'if))
              (cons (list test then) (expand else-part))
              (if (eq? else-part #f)
                  (list (list test then))
                  (list (list test then) (list 'else else-part)))))
        '()))
  (cons 'cond (expand expr)))
