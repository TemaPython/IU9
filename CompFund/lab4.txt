```scheme
(define assertion-exit #f)

(define-syntax assert
  (syntax-rules ()
    ((assert cond)
     (if (not cond)
         (begin
           (display "FAILED: ")
           (write 'cond)
           (newline)
           (assertion-exit))))))


(define (use-assertions)
  (call-with-current-continuation
   (lambda (z)
     (set! assertion-exit z)
     'assertions-enabled)))

(define (proc-desugar source dest)
  (define (transform-definition def)
    (cond
      ((and (pair? def) (equal? (car def) 'define) (pair? (cadr def)))
       (let ((func-name (caadr def)))
         (list 'define func-name (list 'lambda (cdadr def) (caddr def)))))
      (else def)))


    (with-input-from-file source
      (lambda ()
        (with-output-to-file dest
          (lambda ()
            (let loop ((expr (read)))
              (cond
                ((eof-object? expr) #t)
                (else
                 (write (transform-definition expr))
                 (newline)
                 (loop (read))))))))))

(define (text-stat source)
  (with-input-from-file source
    (lambda ()
      (let loop ((ch (read-char)) (isnumber? #f) (dig 0) (punct 0) (isword? #f) (w 0))
        (cond
          ((eof-object? ch) 
           (list w dig punct))

          ((char-word? ch)
           (if isword?
               (loop (read-char) #f dig punct #t w)
               (loop (read-char) #f dig punct #t (+ w 1))))
          
          ((char-numeric? ch)
           (if isnumber?
               (loop (read-char) #t dig punct #f w)
               (loop (read-char) #t (+ dig 1) punct #f w)))
          
          ((char-punctuation? ch)
           (loop (read-char) #f dig (+ punct 1) #f w))
          
          (else
           (loop (read-char) #f dig punct #f w)))))))

(define (char-word? ch)
  (or (char-alphabetic? ch)
      (char=? ch #\-)
      (char=? ch #\')))

(define (char-punctuation? ch)
  (or (char=? ch #\,)
      (char=? ch #\.)
      (char=? ch #\!)
      (char=? ch #\()
      (char=? ch #\))
      (char=? ch #\?)))

(define trib-memo
  (let ((known-results '((0 . 0) (1 . 0) (2 . 1))))
    (lambda (n)
      (let* ((res (assoc n known-results)))
        (if res
            (cdr res)
            (let ((new-res (+ (trib-memo (- n 1))
                              (trib-memo (- n 2))
                              (trib-memo (- n 3)))))
              (set! known-results
                    (cons (cons n new-res) known-results))
              new-res))))))

(define-syntax lazy-cons
  (syntax-rules ()
    ((lazy-cons a b)
     (cons a (delay b)))))

(define (lazy-car p)
  (car p))

(define (lazy-cdr p)
  (force (cdr p)))

(define (lazy-head xs k)
  (if (= k 0)
      '()
      (cons (lazy-car xs) (lazy-head (lazy-cdr xs) (- k 1)))))

(define (lazy-ref xs k)
  (if (= k 0)
      (lazy-car xs)
      (lazy-ref (lazy-cdr xs) (- k 1))))

(define (lazy-map proc xs)
  (lazy-cons (proc (lazy-car xs)) (lazy-map proc (lazy-cdr xs))))

(define (lazy-zip xs ys)
  (lazy-cons (list (lazy-car xs) (lazy-car ys))
             (lazy-zip (lazy-cdr xs) (lazy-cdr ys))))

(define fibonacci
  (letrec ((fib (lazy-cons 1
                           (lazy-cons 1
                                      (lazy-map (lambda (x) (+ (car x) (cadr x)))
                                                (lazy-zip fib (lazy-cdr fib)))))))
    fib))


(define-syntax unless
  (syntax-rules()
    ((unless cond? expr1 expr2 ... exprn)
     (if (not cond?)
         (begin expr1 expr2 ... exprn)))))

(define-syntax while
  (syntax-rules()
    ((while cond? expr1 expr2 ... exprn)
     (letrec ((iter (lambda ()
                      (if cond? 
                          (begin expr1 expr2 ... exprn
                                 (iter))))))
       (iter)))))

(define-syntax cout
  (syntax-rules (<< endl)
    ((cout) (begin))
    
    ((cout << endl . rest)
     (begin (newline) (cout . rest)))
    
    ((cout << expr)
     (display expr))
    
    ((cout << expr . rest)
     (begin (display expr) (cout . rest)))))

```
