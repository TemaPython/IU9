```bash

Файл `reapeat`:
```bash
#!/bin/bash
t="$2"
i=0
"$1" > print$i.txt 2> errors$i.txt &
i=$((i+1))
pid=$!
while true; do
    if kill -0 "$pid"; then
        sleep "$t"m
    else
        "$1" > print$i.txt 2> errors$i.txt &
        i=$((i+1))
        pid=$!
        sleep "$t"m
    fi
done
```
Файл `str`:
```bash
#!/bin/bash
count=0
rec() {
    if [ -d "$1" ]; then
        for name in "$1"/*; do
            rec "$name"
        done
    else
        case "$1" in
            *.c|*.h)
                n=$(grep -cve '^[[:space:]]*$' "$1")
                count=$((count + n))
                ;;
        esac
    fi
}
rec "$1"
echo "$count"
```
Файл `strok.py`:
```bash
#!/usr/bin/env python3
import random
import string
def random_string(ln, cnt):
    alph = string.ascii_letters + string.digits + string.punctuation
    for i in range(cnt):
        st = "".join([random.choice(alph) for i in range(ln)])
        print(st)
```Файл `strok_main.py`:
```bash
#!/usr/bin/env python3
from strok import random_string
import sys
def main():
    length = int(sys.argv[1])
    count = int(sys.argv[2])
    random_string(length, count)

if __name__ == "__main__":
    main()
```
Файл `mynl.py`:
```bash
#!/usr/bin/env python3
import sys
def mynl(stream):
    line_no = 1
    for line in stream:
        if line.strip() == "":
            sys.stdout.write(line)
        else:
            sys.stdout.write(f"{line_no}\t{line}")
            line_no += 1

def main():
    if len(sys.argv) == 1:
        mynl(sys.stdin)
    else:
        for name in sys.argv[1:]:
            with open(name, "r") as f:
                mynl(f)
                
if __name__ == "__main__":
    main()
```
